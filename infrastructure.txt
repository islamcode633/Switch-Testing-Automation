Steps for creating br0:
	apt install bridge-utils
	brctl addbr br0
	ip a
	ip addr add 0.0.0.0 dev enp4s0 	ip link set dev enp4s0 promisc on

	brctl addif br0 enp4s0
	ip addr add 192.168.127.10/24  dev br0
	route add all interface in br0
